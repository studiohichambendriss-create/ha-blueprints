condition:
  - condition: template
    value_template: >-
      {% set o = trigger.event.data.old_state %}
      {% set n = trigger.event.data.new_state %}
      {{ o is not none and n is not none
         and o.state not in ['unavailable','unknown']
         and n.state not in ['unavailable','unknown'] }}

variables:
  targets_entities_ch1: !input scroll_wheel_target_ch1
  targets_entities_ch2: !input scroll_wheel_target_ch2
  targets_entities_ch3: !input scroll_wheel_target_ch3

  scroll_wheel_mode_ch1: !input scroll_wheel_mode_ch1
  scroll_wheel_mode_ch2: !input scroll_wheel_mode_ch2
  scroll_wheel_mode_ch3: !input scroll_wheel_mode_ch3

  var_min_pct: !input dim_min_pct
  var_dim_step_pct: !input dim_step_pct
  var_color_hue_step: !input color_hue_step
  var_color_saturation: !input color_saturation
  var_min_color_temp: !input min_color_temp
  var_max_color_temp: !input max_color_temp
  var_color_temp_step: !input color_temp_step
  var_on_hold_delay: !input on_hold_delay
  var_volume_step_pct: !input volume_step_pct
  var_volume_max_pct: !input volume_max_pct

action:
  - variables:
      channel: >-
        {% if trigger.id in ['press_ch1','scroll_left_ch1','scroll_right_ch1'] %}ch1
        {% elif trigger.id in ['press_ch2','scroll_left_ch2','scroll_right_ch2'] %}ch2
        {% elif trigger.id in ['press_ch3','scroll_left_ch3','scroll_right_ch3'] %}ch3
        {% else %}unknown{% endif %}

      is_press: "{{ trigger.id is match('^press_') }}"
      is_scroll: "{{ trigger.id is match('^scroll_') }}"

      scroll_dir: >-
        {% if trigger.id in ['scroll_left_ch1','scroll_left_ch2','scroll_left_ch3'] %}left
        {% elif trigger.id in ['scroll_right_ch1','scroll_right_ch2','scroll_right_ch3'] %}right
        {% else %}none{% endif %}

      # Alleen reageren op opgaande flank (0->1 of off->on)
      pulse_rising: >-
        {% set o = trigger.event.data.old_state.state %}
        {% set n = trigger.event.data.new_state.state %}
        {{ (o in ['0','off'] and n in ['1','on']) }}

      scroll_steps: "{{ 1 if (is_scroll and pulse_rising) else 0 }}"

      press_event_type: >-
        {% if is_press %}
          {{ trigger.event.data.new_state.attributes.event_type | default('') }}
        {% else %}
          {{ '' }}
        {% endif %}

  - choose:
      ##########################################################################
      # BUTTONS (press)
      ##########################################################################
      # Press: single click
      - conditions:
          - condition: template
            value_template: "{{ is_press and press_event_type == 'multi_press_1' }}"
        sequence:
          - choose:
              - conditions: "{{ channel == 'ch1' }}"
                sequence: !input click_action_ch1
              - conditions: "{{ channel == 'ch2' }}"
                sequence: !input click_action_ch2
              - conditions: "{{ channel == 'ch3' }}"
                sequence: !input click_action_ch3

      # Press: double click
      - conditions:
          - condition: template
            value_template: "{{ is_press and press_event_type == 'multi_press_2' }}"
        sequence:
          - choose:
              - conditions: "{{ channel == 'ch1' }}"
                sequence: !input double_click_action_ch1
              - conditions: "{{ channel == 'ch2' }}"
                sequence: !input double_click_action_ch2
              - conditions: "{{ channel == 'ch3' }}"
                sequence: !input double_click_action_ch3

      # Press: long press + on-hold repeat
      - conditions:
          - condition: template
            value_template: "{{ is_press and press_event_type == 'long_press' }}"
        sequence:
          - choose:
              - conditions: "{{ channel == 'ch1' }}"
                sequence:
                  - sequence: !input long_click_action_ch1
                  - repeat:
                      while: >-
                        {{ repeat.index < 500
                           and state_attr(var_press_event_ch1, 'event_type') == 'long_press' }}
                      sequence:
                        - sequence: !input on_hold_action_ch1
                        - delay: "{{ var_on_hold_delay }}"
              - conditions: "{{ channel == 'ch2' }}"
                sequence:
                  - sequence: !input long_click_action_ch2
                  - repeat:
                      while: >-
                        {{ repeat.index < 500
                           and state_attr(var_press_event_ch2, 'event_type') == 'long_press' }}
                      sequence:
                        - sequence: !input on_hold_action_ch2
                        - delay: "{{ var_on_hold_delay }}"
              - conditions: "{{ channel == 'ch3' }}"
                sequence:
                  - sequence: !input long_click_action_ch3
                  - repeat:
                      while: >-
                        {{ repeat.index < 500
                           and state_attr(var_press_event_ch3, 'event_type') == 'long_press' }}
                      sequence:
                        - sequence: !input on_hold_action_ch3
                        - delay: "{{ var_on_hold_delay }}"

      ##########################################################################
      # SCROLL (WiZ-proof)
      ##########################################################################
      - conditions:
          - condition: template
            value_template: "{{ is_scroll and scroll_steps > 0 and scroll_dir in ['left','right'] }}"
        sequence:
          - repeat:
              for_each: >-
                {{ (targets_entities_ch1 if channel == 'ch1'
                    else (targets_entities_ch2 if channel == 'ch2'
                    else targets_entities_ch3)) }}
              sequence:
                - variables:
                    ent: "{{ repeat.item }}"
                    ent_domain: "{{ repeat.item.split('.')[0] }}"
                    mode: >-
                      {{ (scroll_wheel_mode_ch1 if channel == 'ch1'
                          else (scroll_wheel_mode_ch2 if channel == 'ch2'
                          else scroll_wheel_mode_ch3)) }}
                    step_pct: "{{ (var_dim_step_pct | float(0)) * (scroll_steps | float(1)) }}"
                    cur_bri_pct: >-
                      {% set b = state_attr(ent,'brightness') %}
                      {{ ((b | float(0)) * 100 / 255) | float(0) }}
                    cur_ct: >-
                      {{ state_attr(ent,'color_temp_kelvin') | int( (var_min_color_temp + var_max_color_temp) / 2 ) }}
                    cur_hue: >-
                      {% set hs = state_attr(ent,'hs_color') %}
                      {{ (hs[0] if hs is not none else 0) | float(0) }}
                    is_on: "{{ is_state(ent, 'on') }}"
                    color_mode: "{{ state_attr(ent, 'color_mode') | default('') }}"

                - choose:
                    # LIGHTS: DIM
                    - conditions:
                        - condition: template
                          value_template: "{{ ent_domain == 'light' and mode == 'lights: dim' }}"
                      sequence:
                        - choose:
                            - conditions:
                                - condition: template
                                  value_template: "{{ scroll_dir == 'right' }}"
                              sequence:
                                - action: light.turn_on
                                  target:
                                    entity_id: "{{ ent }}"
                                  data:
                                    brightness_pct: >-
                                      {% if not is_on %}
                                        {{ [var_min_pct, step_pct] | max }}
                                      {% else %}
                                        {{ [100, (cur_bri_pct + step_pct)] | min }}
                                      {% endif %}
                                    transition: 0.1

                            - conditions:
                                - condition: template
                                  value_template: "{{ scroll_dir == 'left' and is_on }}"
                              sequence:
                                - action: light.turn_on
                                  target:
                                    entity_id: "{{ ent }}"
                                  data:
                                    brightness_pct: >-
                                      {{ [var_min_pct, (cur_bri_pct - step_pct)] | max }}
                                    transition: 0.1

                    # LIGHTS: COLOR TEMP (of mixed-mode wanneer color_mode == color_temp)
                    - conditions:
                        - condition: template
                          value_template: >-
                            {{ ent_domain == 'light' and (
                                 mode == 'lights: color temp only'
                                 or (mode == 'lights: color temp and color hue' and color_mode == 'color_temp')
                               ) }}
                      sequence:
                        - action: light.turn_on
                          target:
                            entity_id: "{{ ent }}"
                          data:
                            color_temp_kelvin: >-
                              {% set delta = (var_color_temp_step | int(0)) * (scroll_steps | int(1)) %}
                              {% set wanted = (cur_ct + (delta if scroll_dir == 'right' else -delta)) %}
                              {{ [var_max_color_temp, [var_min_color_temp, wanted] | max] | min }}

                    # LIGHTS: COLOR HUE (of mixed-mode wanneer NIET color_temp)
                    - conditions:
                        - condition: template
                          value_template: >-
                            {{ ent_domain == 'light' and (
                                 mode == 'lights: color hue only'
                                 or (mode == 'lights: color temp and color hue' and color_mode != 'color_temp')
                               ) }}
                      sequence:
                        - action: light.turn_on
                          target:
                            entity_id: "{{ ent }}"
                          data:
                            hs_color: >-
                              {% set delta = (var_color_hue_step | float(0)) * (scroll_steps | float(1)) %}
                              {% set raw = (cur_hue + (delta if scroll_dir == 'right' else -delta)) %}
                              {% set hue = (raw % 360) %}
                              {{ [hue, (var_color_saturation | float(100))] }}

                    # MEDIA: VOLUME
                    - conditions:
                        - condition: template
                          value_template: "{{ ent_domain == 'media_player' and mode == 'media: volume control' }}"
                      sequence:
                        - action: media_player.volume_set
                          target:
                            entity_id: "{{ ent }}"
                          data:
                            volume_level: >-
                              {% set cur = (state_attr(ent, 'volume_level') | float(0)) * 100 %}
                              {% set delta = (var_volume_step_pct | float(0)) * (scroll_steps | float(1)) %}
                              {% set wanted = cur + (delta if scroll_dir == 'right' else -delta) %}
                              {% set capped = [var_volume_max_pct, wanted] | min %}
                              {{ [[capped, 0] | max, 100] | min / 100 }}
